create database flights;
use flights;

SET FOREIGN_KEY_CHECKS = 0;

DROP TABLE IF EXISTS  `account`;
CREATE TABLE `account` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `name` varchar(50),
  `email` varchar(50) NOT NULL,
  `psd` varchar(50) NOT NULL,
  `createtime` varchar(20) DEFAULT NULL,
  `type` int(1),
  `status` int(1),
  PRIMARY KEY (`id`)
)  DEFAULT CHARSET=utf8;


DROP TABLE IF EXISTS `user`;
CREATE TABLE `user`(
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `accountid` int(11) NOT NULL,
  `realname` varchar(100),
  `address`  varchar(1000),
  `tel`  varchar(50),
  `qq` varchar(20),
  `businesslicenseurl` varchar(500),
  PRIMARY KEY(`id`)
) DEFAULT CHARSET=utf8;

DROP TABLE IF EXISTS `line`;
CREATE TABLE `line`(
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `accountid` int(11) NOT NULL,
  `name` varchar(50),
  `createtime` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`id`)
) DEFAULT CHARSET=utf8;
ALTER TABLE `line` ADD UNIQUE(`name`);


DROP TABLE IF EXISTS `product`;
CREATE TABLE `product`(
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `lineid` int(11) NOT NULL,
  `title` varchar(1000),
  `price` varchar(10),
  `childprice` varchar(10),
  `description` longtext,
  `photourl` varchar(500),
  `promotephotourl` varchar(500),
  `createtime` varchar(20) DEFAULT NULL,
  PRIMARY KEY(`id`)
) DEFAULT CHARSET=utf8;

DROP TABLE IF EXISTS `productdate`;
CREATE TABLE `productdate`(
  `productid` int(11) NOT NULL,
  `productdate` varchar(20) NOT NULL,
  `inventory` int(5),
  `total` int(5),
  PRIMARY KEY(`productid`,`productdate`)
) DEFAULT CHARSET=utf8;

DROP TABLE IF EXISTS `purchaseinfo`;
CREATE TABLE `purchaseinfo`(
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `productid` int(11) NOT NULL,
  `productdate` varchar(20) NOT NULL,
  `realname` varchar(100),
  `birthday` varchar(20),
  `isadult` int(1),
  `accountid` int(11) NOT NULL,
  `cardtype` int(1),
  `cardnumber` varchar(50),
  `cardvalidate` varchar(20),
  `createtime` varchar(20) DEFAULT NULL,
  PRIMARY KEY(`id`)
) DEFAULT CHARSET=utf8;

CREATE TABLE `resetpassword`(
	`userid` int(11) NOT NULL,
	`token`  varchar(50) NOT NULL,
	PRIMARY KEY (`userid`)
)  DEFAULT CHARSET=utf8;

CREATE TABLE `actionlog`(
	`id` int(11) NOT NULL AUTO_INCREMENT,
	`accountid` int(11) NOT NULL,
	`action`  varchar(20),
	`description` varchar(1000),
	`createtime` varchar(20) DEFAULT NULL,
	PRIMARY KEY(`id`)
)  DEFAULT CHARSET=utf8;
